  <script>
    // utilitaires
    const uid = 'arviina_phrases_v1';
    const todayStr = new Date().toISOString().slice(0,10);

    function load() {
      const raw = localStorage.getItem(uid) || '{}';
      try { return JSON.parse(raw); } catch(e){ return {}; }
    }
    function save(obj){ localStorage.setItem(uid, JSON.stringify(obj)); }

    function renderToday(){
      const phrases = load();
      const p = phrases[todayStr];
      const el = document.getElementById('todayPhrase');
      const meta = document.getElementById('todayMeta');
      if(p){ el.textContent = '"' + p + '"'; meta.textContent = 'Pour ' + todayStr; }
      else { el.textContent = 'Aucune phrase définie pour aujourd\'hui — ajoute-en une à droite.'; meta.textContent = ''; }
    }

    function renderList(){
      const list = document.getElementById('list'); list.innerHTML = '';
      const phrases = load();
      const entries = Object.entries(phrases).sort((a,b)=> b[0].localeCompare(a[0]));
      if(entries.length===0){ list.innerHTML = '<small>Aucune phrase enregistrée.</small>'; return; }
      for(const [date,txt] of entries){
        const it = document.createElement('div'); it.className='item';
        const span = document.createElement('div'); span.innerHTML = '<div class="date">'+date+'</div><div style="color:var(--muted);font-size:13px">'+(txt.length>60?txt.slice(0,60)+'…':txt)+'</div>';
        const ctrl = document.createElement('div'); ctrl.className='controls';
        const edit = document.createElement('button'); edit.className='btn-ghost'; edit.textContent='Éditer';
        edit.onclick = ()=>{ document.getElementById('dateInput').value = date; document.getElementById('phraseInput').value = txt; window.scrollTo({top:0,behavior:'smooth'}); };
        const del = document.createElement('button'); del.textContent='Supprimer'; del.onclick = ()=>{ if(confirm('Supprimer la phrase du '+date+' ?')){ const ph = load(); delete ph[date]; save(ph); renderList(); renderToday(); } };
        ctrl.appendChild(edit); ctrl.appendChild(del);
        it.appendChild(span); it.appendChild(ctrl); list.appendChild(it);
      }
    }

    document.getElementById('saveBtn').addEventListener('click', ()=>{
      const date = document.getElementById('dateInput').value.trim();
      const txt = document.getElementById('phraseInput').value.trim();
      if(!/^\d{4}-\d{2}-\d{2}$/.test(date)){ alert('Utilise le format YYYY-MM-DD'); return; }
      const ph = load();
      if(txt===''){ if(confirm('La phrase est vide : veux-tu supprimer la date '+date+' ?')){ delete ph[date]; save(ph); renderList(); renderToday(); } return; }
      ph[date] = txt; save(ph); renderList(); renderToday(); alert('Phrase enregistrée pour '+date);
    });

    document.getElementById('clearBtn').addEventListener('click', ()=>{
      const date = document.getElementById('dateInput').value.trim();
      if(!/^\d{4}-\d{2}-\d{2}$/.test(date)){ alert('Utilise le format YYYY-MM-DD'); return; }
      const ph = load(); if(ph[date]){ if(confirm('Supprimer la phrase du '+date+' ?')){ delete ph[date]; save(ph); renderList(); renderToday(); } } else alert('Aucune phrase pour cette date.');
    });

    // Export
    document.getElementById('exportBtn').addEventListener('click', ()=>{
      const data = localStorage.getItem(uid) || '{}';
      const blob = new Blob([data],{type:'application/json'});
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a'); a.href = url; a.download = 'arviina_phrases.json'; a.click(); URL.revokeObjectURL(url);
    });

    // Import
    document.getElementById('importFile').addEventListener('change',(e)=>{
      const f = e.target.files[0]; if(!f) return; const r = new FileReader(); r.onload = ()=>{
        try{ const parsed = JSON.parse(r.result); if(typeof parsed !== 'object'){ alert('Fichier invalide'); return; } save(parsed); renderList(); renderToday(); alert('Import réussi'); }catch(err){ alert('Erreur pendant l\'import'); }
      }; r.readAsText(f);
    });

    // Init
    document.getElementById('dateInput').value = todayStr;
    renderList(); renderToday();

  </script>
